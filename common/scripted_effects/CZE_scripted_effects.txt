CZE_increase_skoda_max_optimization_tier = { #Moves Skida Optimization up one tier, should only be used four times in the focus tree. Max cap minus starting optimization tier number: 7 - 3 = 4. 
	if = {
		limit = {
			any_core_state = {
				is_controlled_by = ROOT
			}
			has_variable = skoda_max_optimization_tier #Can never be too safe these days
			check_variable = {
				skoda_max_optimization_tier < 7
			}
		}
		add_to_variable = {
			var = skoda_max_optimization_tier
			value = 1
			tooltip = CZE_increase_skoda_max_optimization_tier_var_tt
		}
	}
	else_if = { #Fall back reward if we're above the four uses somehow 
		limit = {
			has_dynamic_modifier = { modifier = CZE_skoda_works_dynamic_modifier }
		}
		custom_effect_tooltip = CZE_modify_skoda_works_ns_intro_tt
		add_to_variable = {
			CZE_skoda_works_dynamic_modifier_production_factory_max_efficiency_factor = 0.01
			tooltip = production_factory_max_efficiency_factor_tt
		}
	}
	else = {
	}
}

CZE_get_rid_of_the_auto_blok = { #Only you can stop immersion breaking continuity errors - Smokey the Bear, probably. 
	if = { #Gets rid of the Auto Blok in context were them being around would be a continuity error. e.a Slovakia existing
		limit = {
			NOT = {
				has_country_flag = CZE_get_rid_of_the_auto_blok_happen_once_already
			}
		}
		ROOT = { #We need ROOT to be extra safe because of a Character effect and a Overlord reference
			if = {
				limit = {
					has_country_leader = {
						character = CZE_andrej_hlinka 
						ruling_only = no
					}
					NOT = {
						OR = {
							if = {
								limit = {
									is_subject = yes
								}
								OVERLORD = {
									has_government = neutrality
								}
							}
						}
					}
				}
				set_party_name = {
					ideology = neutrality
					long_name = CZE_neutrality_party_placeholder
					name = CZE_neutrality_party_placeholder #intentionally duplicate
				}
			}
			if = {
				limit = {
					has_character = CZE_andrej_hlinka 
				}
				retire_character = CZE_andrej_hlinka
			}
			set_country_flag = CZE_get_rid_of_the_auto_blok_happen_once_already
		}
	}
}
