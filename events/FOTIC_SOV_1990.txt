country_event = {
	id = FOTIC_SOV_1990.100
	hidden = yes
	is_triggered_only = yes

	immediate = {
		add_to_variable = { fotic_sov_deficit = 1 }
		add_to_variable = { fotic_sov_party_discipline = -1 }
		add_to_variable = { fotic_sov_republican_separatism = 1 }
		add_to_variable = { fotic_sov_street_pressure = 1 }
		add_to_variable = { fotic_sov_black_market = 1 }

		if = {
			limit = {
				check_variable = { fotic_sov_deficit > 70 }
			}
			add_to_variable = { fotic_sov_street_pressure = 1 }
			add_to_variable = { fotic_sov_black_market = 1 }
		}

		if = {
			limit = {
				check_variable = { fotic_sov_party_discipline < 40 }
			}
			add_to_variable = { fotic_sov_republican_separatism = 1 }
			add_to_variable = { fotic_sov_russian_challenge = 1 }
		}

		if = {
			limit = {
				check_variable = { fotic_sov_kgb_influence > 65 }
			}
			add_to_variable = { fotic_sov_conservative_pressure = 1 }
			add_to_variable = { fotic_sov_reformist_influence = -1 }
		}

		if = {
			limit = {
				check_variable = { fotic_sov_army_politicization > 50 }
			}
			add_to_variable = { fotic_sov_gkchp_risk = 1 }
		}

		FOTIC_SOV_clamp_crisis_variables = yes
	}

	option = {
		name = FOTIC_SOV_1990.100.a
	}
}
